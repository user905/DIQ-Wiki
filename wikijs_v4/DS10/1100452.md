CREATE FUNCTION [dbo].[fnDIQ_DS10_CCLogDetails_AreAUWDollarsNegativeAndNTENonZero] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	
	SELECT 
		*
	FROM 
		DS10_CC_log_detail
	WHERE 
		upload_ID = @upload_ID
	  AND AUW = 'Y'
	  AND dollars_delta < 0

	  AND NTE_dollars_delta <> 0
