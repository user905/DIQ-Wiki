CREATE FUNCTION [dbo].[fnDIQ_DS06_Res_IsResourceAndRoleIDMissing] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	
	SELECT
		*
	FROM
		DS06_schedule_resources

			upload_id = @upload_ID
