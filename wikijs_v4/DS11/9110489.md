CREATE FUNCTION [dbo].[fnDIQ_DS11_VAR_IsPPVARMisalignedWithDS01Type] (
	@upload_id int = 0

RETURNS TABLE
AS RETURN
(
	
	SELECT
		V.*
	FROM 
		DS11_variance V INNER JOIN DS01_WBS W ON V.WBS_ID = W.WBS_ID
