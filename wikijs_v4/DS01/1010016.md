CREATE FUNCTION [dbo].[fnDIQ_DS01_WBS_IsCAorSLPPAtLowerLevelThanWP] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	
	SELECT 
		* 
	FROM 
		DS01_WBS 
	WHERE 
		upload_ID = @upload_ID
	AND	WBS_ID IN (
			SELECT WBS_ID 
			FROM AncestryTree_Get(@upload_ID) A
