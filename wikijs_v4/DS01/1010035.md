CREATE FUNCTION [dbo].[fnDIQ_DS01_WBS_IsWPMFoundOnNonPPorWP] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	SELECT 
		*
	FROM
		DS01_WBS
	WHERE
			upload_ID = @upload_ID
		AND ISNULL(WPM,'') <> ''
		AND type NOT IN ('WP','PP')
)
