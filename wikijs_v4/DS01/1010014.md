## Code

```sql

CREATE FUNCTION [dbo].[fnDIQ_DS01_WBS_IsCAMInconsistentInCABranch] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	
	SELECT 
		*
	FROM
		DS01_WBS
	WHERE
			upload_ID = @upload_ID
		AND WBS_ID IN (
			SELECT WBS_ID
			FROM AncestryTree_Get(@upload_ID)
			WHERE Ancestor_Type = 'CA' AND CAM <> Ancestor_CAM
		)
)
```
