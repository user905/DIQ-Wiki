CREATE FUNCTION [dbo].[fnDIQ_DS04_Sched_IsHardConstraintMissingDate] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	SELECT
		*
	FROM
		DS04_schedule
	WHERE
			upload_id = @upload_ID
		AND constraint_type in (SELECT type FROM HardConstraints)
		AND constraint_date IS NULL
)
