CREATE FUNCTION [dbo].[fnDIQ_DS04_Sched_IsWBSMissingInDS03] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	
	SELECT
		*
	FROM
		DS04_schedule
	WHERE
			upload_id = @upload_ID
		AND TRIM(ISNULL(subtype,'')) = ''
		AND WBS_ID NOT IN (SELECT WBS_ID_WP FROM DS03_cost WHERE upload_ID = @upload_ID)
)
