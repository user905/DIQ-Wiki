CREATE FUNCTION [dbo].[fnDIQ_DS04_Sched_IsLOETypeMismatchedWithEVT] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	
	SELECT
		*
	FROM
		DS04_schedule
	WHERE
			upload_id = @upload_ID
		AND type = 'LOE'
		AND TRIM(ISNULL(EVT,'')) NOT IN ('A','')
)
