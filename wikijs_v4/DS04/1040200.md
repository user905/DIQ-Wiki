CREATE FUNCTION [dbo].[fnDIQ_DS04_Sched_IsPCGT0MissingASDate] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	
	SELECT
		*
	FROM
		DS04_schedule
	WHERE
			upload_id = @upload_ID
		AND schedule_type = 'FC'
		AND (PC_duration > 0 OR PC_units > 0 OR PC_physical > 0)
		AND AS_date IS NULL
)
