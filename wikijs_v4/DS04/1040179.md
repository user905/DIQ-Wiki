CREATE FUNCTION [dbo].[fnDIQ_DS04_Sched_IsFinishedTaskMissingActualStart] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
		
	SELECT
		*
	FROM
		DS04_schedule
	WHERE
			upload_id = @upload_ID
		AND schedule_type = 'FC'
		AND AF_date IS NOT NULL
		AND AS_date IS NULL
		AND type != 'SM'
)
