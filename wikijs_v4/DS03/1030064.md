## Code

```sql

CREATE FUNCTION [dbo].[fnDIQ_DS03_Cost_DoesETCExistInThePast] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	SELECT 
		* 
	FROM 
		DS03_Cost
	WHERE
			upload_ID = @upload_ID
		AND (ETCi_dollars <> 0 OR ETCi_FTEs <> 0 OR ETCi_hours <> 0)
		AND period_date <= CPP_status_date 
)
```
