CREATE FUNCTION [dbo].[fnDIQ_DS03_Cost_DoesBCWPExistOnNonWPData] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	
	SELECT 
		* 
	FROM 
		DS03_Cost
	WHERE
			upload_ID = @upload_ID
		AND (BCWPi_dollars <> 0 OR BCWPi_FTEs <> 0 OR BCWPi_hours <> 0)
		AND (TRIM(ISNULL(WBS_ID_WP,'')) = '' OR EVT = 'K')
)
