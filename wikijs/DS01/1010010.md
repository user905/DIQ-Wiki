CREATE FUNCTION [dbo].[fnDIQ_DS01_WBS_DoesWPMMatchCAM] (
	@upload_id int = 0

RETURNS TABLE
AS RETURN
(
	SELECT 
		*
	FROM
		DS01_WBS
