CREATE FUNCTION [dbo].[fnDIQ_DS06_Res_IsResourceMissingEOC] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	SELECT *
	FROM DS06_schedule_resources
	WHERE upload_id = @upload_ID AND ISNULL(resource_ID,'') <> '' AND ISNULL(eoc,'') = ''
)
