## Code

```sql

CREATE FUNCTION [dbo].[fnDIQ_DS08_WAD_IsEVTMissing] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	
	SELECT 
		*
	FROM
		DS08_WAD
	WHERE
			upload_ID = @upload_ID   
		AND TRIM(ISNULL(WBS_ID_WP,'')) <> ''
		AND TRIM(ISNULL(EVT,'')) = ''
)
```
