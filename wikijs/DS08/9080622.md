CREATE FUNCTION [dbo].[fnDIQ_DS08_WAD_IsWPWBSMissingInDS01] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	
	SELECT 
		*
	FROM
		DS08_WAD
