CREATE FUNCTION [dbo].[fnDIQ_DS09_CCLog_IsCCLogMissingInDS10] (
	@upload_id int = 0

RETURNS TABLE
AS RETURN
(
	
	SELECT 
		*
	FROM
		DS09_CC_log
