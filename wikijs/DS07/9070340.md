```sql

CREATE FUNCTION [dbo].[fnDIQ_DS07_IPMR_DoesOTBOTSDateExistWithoutDS04Rpg] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	
	SELECT 
		*
	FROM
		DS07_IPMR_header
	WHERE
			upload_ID = @upload_ID
		AND (SELECT COUNT(*) FROM DS04_schedule	WHERE upload_ID = @upload_ID AND RPG = 'Y') = 0
		AND OTB_OTS_date IS NOT NULL
)
```
