CREATE FUNCTION [dbo].[fnDIQ_DS04_Sched_IsMSLevelDescriptionInsufficient] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	
	SELECT 
		*
	FROM
		DS04_schedule
	WHERE
			upload_ID = @upload_ID
		AND CHARINDEX(' ',TRIM([milestone_level_description])) = 0
)
