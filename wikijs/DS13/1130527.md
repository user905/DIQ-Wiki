## Code

```sql

CREATE FUNCTION [dbo].[fnDIQ_DS13_SubK_IsCostCompleteSubKMissingAFDate] (
	@upload_id int = 0
)
RETURNS TABLE
AS RETURN
(
	
	SELECT
		*
	FROM 
		DS13_subK
	WHERE 
			upload_ID = @upload_ID 
		AND BCWPc_dollars / NULLIF(BAC_dollars,0) = 1
		AND AF_date IS NULL
)
```
